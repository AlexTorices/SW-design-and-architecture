def undo
  if (@speed == CeilingFan::OFF)
    @fan.off
  elsif (@speed == CeilingFan::LOW)
    @fan.low
  elsif (@speed == CeilingFan::MEDIUM)
    @fan.medium
  else
    @fan.high
  end
end
